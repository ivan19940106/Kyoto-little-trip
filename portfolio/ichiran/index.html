//event bus
var bus = new Vue();

Vue.component('kaedama', {
    template: '',
    methods: {
        reset: function(){
            if($('.ingradients').hasClass('eaten-up')){
                bus.$emit('reset');
                $('.ingradients').removeClass('eaten-up');
            }
        }
    }
});

var vm = new Vue({
    el: '#app',
    data: {
        negis: 100,
        meats: 2,
        kara: 70,
        ordered: false
    },
    methods: {
        reset(){
            this.negis = 0;
            this.meats = 0;
            this.kara = 0;
        }
    },
    created: function(){
        bus.$on('reset', this.reset);
    },
    beforeDestroy: function(){
        bus.$off('reset', this.reset);
    }
});
